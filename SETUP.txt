# VIDEO OPS — 完全セットアップガイド
# このファイルを読みながら上から順に実行してください

# ============================================================
# STEP 0: このガイドについて
# ============================================================
#
# VIDEO OPS に含まれる機能:
#
#   ✅ ログイン・認証        Google / メール+パスワード / 招待制
#   ✅ 案件管理              案件ID自動採番・SNS URL・ランク別単価・メモタイムライン
#   ✅ 納品物管理            ステータス管理・担当者紐付け
#   ✅ メンバー管理          編集者ランクA/B/C（追加・削除自由）
#   ✅ 素材エディター        動画プレビュー・タイムスタンプコメント・ナレッジ昇格
#   ✅ AI動画解析            アップロード→自動リネーム→内容レポート生成
#   ✅ ナレッジベース        コメントをAIで構造化・週次レポート
#   ✅ 請求書発行            納品済みのみ・消費税計算・PDF印刷
#   ✅ チーム管理            招待リンク発行・権限管理（管理者のみ）
#
# 所要時間: 約10〜15分（Node.jsインストール済みの場合は5分）
#
# ============================================================


# ============================================================
# STEP 1: Node.js のインストール確認
# ============================================================
#
# ターミナル（Terminal.app）を開いて以下を実行:

node -v

# ✅ v18.x.x 以上が表示されれば OK → STEP 2へ
# ❌ command not found と出た場合 → 以下を実行

/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
brew install node

# インストール後に再確認
node -v   # v22.x.x などが表示されればOK
npm -v    # 10.x.x などが表示されればOK


# ============================================================
# STEP 2: ファイルを配置する
# ============================================================
#
# ダウンロードした video-ops フォルダを
# 例: デスクトップ に置いてください
#
# フォルダ構成の確認:
#   video-ops/
#   ├── server.js        ← サーバー本体
#   ├── auth.js          ← 認証（Google / パスワード）
#   ├── package.json     ← パッケージ定義
#   ├── .env.example     ← 環境変数テンプレート
#   ├── db/
#   │   └── db.js        ← データベース管理
#   └── public/
#       ├── index.html   ← メイン画面
#       ├── editor.html  ← 素材エディター
#       ├── login.html   ← ログイン画面
#       └── invite.html  ← 招待登録画面


# ============================================================
# STEP 3: フォルダに移動してパッケージをインストール
# ============================================================

# デスクトップに置いた場合:
cd ~/Desktop/video-ops

# パッケージ一括インストール（初回のみ・1〜3分かかります）
npm install

# ✅ added xxx packages と表示されれば成功
# ❌ エラーが出た場合 → エラー文をそのままコピーしてClaudeに貼ってください


# ============================================================
# STEP 4: 環境変数ファイルを設定する
# ============================================================

# .env.example をコピーして .env を作成
cp .env.example .env

# .env をテキストエディタで開く（以下のどちらかで開けます）
open -e .env           # TextEdit で開く
# または
open .env              # 関連付けられたアプリで開く

# .env の中身を以下のように編集してください:
# ---------------------------------------------------

PORT=3000

# 【必須】セッション管理用の秘密鍵
# 任意の長い文字列を設定してください（英数字なら何でも可）
SESSION_SECRET=videoops2025myteamsecret

# 【必須】あなたのメールアドレス
# このアドレスで初回ログインすると自動で「管理者」になります
ADMIN_EMAIL=your.name@gmail.com

# 【任意】AI機能（動画解析・ナレッジ生成）を使う場合
# https://console.anthropic.com/ でAPIキーを取得して設定
ANTHROPIC_API_KEY=sk-ant-xxxxxxxxxx

# 【任意】Google ログインを使う場合
# ※ メール+パスワードだけでも全機能使えます
# 設定方法は STEP 5 を参照
GOOGLE_CLIENT_ID=
GOOGLE_CLIENT_SECRET=

APP_URL=http://localhost:3000
DB_PATH=./db/videoops.db

# ---------------------------------------------------
# 編集後に保存してください


# ============================================================
# STEP 5: サーバーを起動する
# ============================================================

npm start

# ✅ 以下のような表示が出れば成功:
#
#   ╔═══════════════════════════════════════╗
#   ║   VIDEO OPS サーバー起動              ║
#   ║   http://localhost:3000               ║
#   ╚═══════════════════════════════════════╝
#
# ブラウザで以下を開いてください:
# → http://localhost:3000
#
# ログイン画面が表示されます。
# 「メールアドレスでログイン」から .env に設定したメールアドレスで
# パスワードを新規設定して入ってください。
#
# ⚠️ 注意: ターミナルを閉じるとサーバーも止まります
#          使う間はターミナルを開いたままにしてください


# ============================================================
# STEP 6: 初回起動後の設定
# ============================================================
#
# 1. ブラウザで http://localhost:3000 を開く
# 2. ログイン画面で ADMIN_EMAIL に設定したアドレスを入力
#    ※ 初回のみ: 任意のパスワードを設定してログイン
# 3. 左メニューの「チーム管理」から:
#    - 編集者のランクを確認（デフォルト: A / B / C）
#    - チームメンバーを招待（招待リンクをコピーして共有）
# 4. 「案件管理」から最初の案件を登録


# ============================================================
# STEP 7（任意）: Google ログインの設定
# ============================================================
#
# Googleアカウントでのログインを有効にする場合の手順:
#
# 1. https://console.cloud.google.com/ にアクセス
# 2. 新規プロジェクトを作成（例: "video-ops"）
# 3. 左メニュー「APIとサービス」→「認証情報」
# 4. 「認証情報を作成」→「OAuthクライアントID」
# 5. アプリケーションの種類: 「ウェブアプリケーション」
# 6. 承認済みリダイレクトURIに追加:
#    http://localhost:3000/auth/google/callback
# 7. 作成されたクライアントID・シークレットを .env に設定:
#    GOOGLE_CLIENT_ID=xxxx.apps.googleusercontent.com
#    GOOGLE_CLIENT_SECRET=GOCSPX-xxxx
# 8. npm start で再起動するとGoogleログインボタンが有効になります


# ============================================================
# STEP 8（任意）: AI機能の設定
# ============================================================
#
# 動画の自動解析・ナレッジ生成・週次レポートを使う場合:
#
# 1. https://console.anthropic.com/ にアクセス（Anthropicアカウント作成）
# 2. 「API Keys」→「Create Key」でAPIキーを取得
# 3. .env の ANTHROPIC_API_KEY に設定
# 4. npm start で再起動
#
# ※ APIキーがなくても以下は使えます:
#    ✅ 案件管理・納品物管理・請求書
#    ✅ ログイン・メンバー招待
#    ✅ 素材アップロード・リネーム（AI解析なし）
#    ✅ タイムスタンプコメント
#    ❌ 動画内容の自動解析レポート
#    ❌ ナレッジのAI生成
#    ❌ 週次レポートのAI分析


# ============================================================
# STEP 9（任意）: 動画解析を強化する場合（ffmpeg）
# ============================================================
#
# ffmpeg（エフエフエムペグ）: 動画からフレーム画像を抽出するツール
# 入っているとAIが動画の映像を「見て」解析できるようになります
# 入っていなくても動作はします（テキスト解析のみになります）
#
brew install ffmpeg

# 確認
ffmpeg -version


# ============================================================
# よくあるトラブル
# ============================================================
#
# ● 「Error: Cannot find module 'better-sqlite3'」
#   → npm install をもう一度実行してください
#
# ● 「Error: listen EADDRINUSE :::3000」
#   → 3000番ポートが使用中です。以下で止めてから再起動:
#   lsof -ti:3000 | xargs kill
#
# ● ブラウザで「このサイトにアクセスできません」
#   → ターミナルでサーバーが起動しているか確認してください
#   → npm start が実行中かどうか確認
#
# ● ログインできない
#   → .env の ADMIN_EMAIL が正しいか確認
#   → .env を編集後は npm start で再起動が必要
#
# ● npm install でエラーが出る
#   → node -v で Node.js v18以上か確認
#   → sudo npm install で試す

cd ~/Desktop/video-ops

npm run dev
